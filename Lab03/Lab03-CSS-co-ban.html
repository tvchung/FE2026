<h1>Học CSS cơ bản</h1>

<p id="special">Đây là đoạn văn có id special.</p>
<div class="box">
  <p>Đoạn văn bên trong div.box</p>
</div>
<h2>Danh sách môn học</h2>
<ul>...</ul>

/* 1. Bộ chọn theo thẻ */
p { color: blue; font-size: 16px; }
/* 2. Bộ chọn theo class */
.highlight { background-color: yellow; font-weight: bold; }
/* 3. Bộ chọn theo id */
#special { color: red; font-size: 18px; }
/* 4. Bộ chọn hậu duệ */
.box p { color: green; }
/* 5. Bộ chọn nhóm */
h1, h2 { color: purple; }
/* 6. Trang trí thêm */
ul li { color: orange; }
