Switch Case - Hiển thị thứ trong tuần

        var day = 3;
        switch (day) {
            case 1:
                document.write("Thứ hai");
                break;
            case 2:
                document.write("Thứ ba");
                break;
            case 3:
                document.write("Thứ tư");
                break;
            default:
                document.write("Không hợp lệ");
        }
